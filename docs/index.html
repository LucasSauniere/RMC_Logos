üöÄ Quick Start

Installation

pip install rmclogo

Basic Usage

from rmclogo import RidgeMap

# Create a ridge map for Marseille
rm = RidgeMap(bbox=(5.29, 43.22, 5.45, 43.37))
values = rm.get_elevation_data(num_lines=80, elevation_pts=300)

# Save as PNG
fig, ax = plt.subplots(figsize=(12, 16))
rm.plot_map(values=values, label='MARSEILLE', ax=ax)
plt.savefig('marseille.png', dpi=300)

# Or save as SVG
rm.save_svg(
    output_path='marseille.svg',
    values=values,
    label='MARSEILLE',
    line_color='black',
    background_color='#ffffff'
)
